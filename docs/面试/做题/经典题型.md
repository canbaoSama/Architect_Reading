## this 指向

```js
let length = 20
let obj = {
    length: 100,
    sayHi() {
        console.log(this.length)
    },
}
obj.sayHi() // 100
const fn = obj.sayHi
fn() // 20（非严格下 window.length）

const arr = [obj.sayHi, 2, 3, 4, 5]
arr[0]() // 5（数组长度）
```

## 重写 toString

```js
let a = {
    i: 1,
    toString() { return a.i++ },
}
console.log(a == 1 && a == 2 && a == 3) // true

let b = [1, 2, 3]
b.toString = b.shift
console.log(b == 1 && b == 2 && b == 3) // true
```
